  kind: VaultRestart
  metadata:
    name: example-vault-restart
    namespace: {{ .Release.Namespace }}
  spec:
    statefulSetName: vault
    secretName: vault-tls-secret
    vaultAddress: "https://vault.vault.svc:8200"
    vaultRole: "vault-operations-role"
    reason: cert-rotation
    dryRun: false
  EOF
